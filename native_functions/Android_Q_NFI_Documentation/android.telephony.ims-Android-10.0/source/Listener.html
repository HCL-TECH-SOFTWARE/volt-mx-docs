/**@class android.telephony.ims.ImsCallSession.Listener
@extends java.lang.Object

 Listener for events relating to an IMS session, such as when a session is being
 recieved ("on ringing") or a call is outgoing ("on calling").
 <p>Many of these events are also received by {@link ImsCall.Listener}.</p>
 @hide
*/
var Listener = {

/**Called when a request is sent out to initiate a new session
 and 1xx response is received from the network.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionProgressing : function(  ) {},

/**Called when the session is established.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionStarted : function(  ) {},

/**Called when the session establishment is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the session establishment failure
*/
callSessionStartFailed : function(  ) {},

/**Called when the session is terminated.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the session termination
*/
callSessionTerminated : function(  ) {},

/**Called when the session is in hold.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionHeld : function(  ) {},

/**Called when the session hold is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the session hold failure
*/
callSessionHoldFailed : function(  ) {},

/**Called when the session hold is received from the remote user.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionHoldReceived : function(  ) {},

/**Called when the session resume is done.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionResumed : function(  ) {},

/**Called when the session resume is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the session resume failure
*/
callSessionResumeFailed : function(  ) {},

/**Called when the session resume is received from the remote user.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionResumeReceived : function(  ) {},

/**Called when the session merge has been started.  At this point, the {@code newSession}
 represents the session which has been initiated to the IMS conference server for the
 new merged conference.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsCallSession}} newSession the session object that is merged with an active & hold session
*/
callSessionMergeStarted : function(  ) {},

/**Called when the session merge is successful and the merged session is active.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionMergeComplete : function(  ) {},

/**Called when the session merge has failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the call merge failure
*/
callSessionMergeFailed : function(  ) {},

/**Called when the session is updated (except for hold/unhold).
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionUpdated : function(  ) {},

/**Called when the session update is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the session update failure
*/
callSessionUpdateFailed : function(  ) {},

/**Called when the session update is received from the remote user.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionUpdateReceived : function(  ) {},

/**Called when the session is extended to the conference session.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsCallSession}} newSession the session object that is extended to the conference
      from the active session
*/
callSessionConferenceExtended : function(  ) {},

/**Called when the conference extension is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference extension failure
*/
callSessionConferenceExtendFailed : function(  ) {},

/**Called when the conference extension is received from the remote user.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionConferenceExtendReceived : function(  ) {},

/**Called when the invitation request of the participants is delivered to the conference
 server.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionInviteParticipantsRequestDelivered : function(  ) {},

/**Called when the invitation request of the participants is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference invitation failure
*/
callSessionInviteParticipantsRequestFailed : function(  ) {},

/**Called when the removal request of the participants is delivered to the conference
 server.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionRemoveParticipantsRequestDelivered : function(  ) {},

/**Called when the removal request of the participants is failed.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
@param {Object {ImsReasonInfo}} reasonInfo detailed reason of the conference removal failure
*/
callSessionRemoveParticipantsRequestFailed : function(  ) {},

/**Called when the conference state is updated.
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionConferenceStateUpdated : function(  ) {},

/**Called when the USSD message is received from the network.
@param {Object {ImsCallSession}} mode mode of the USSD message (REQUEST / NOTIFY)
@param {Number} ussdMessage USSD message
*/
callSessionUssdMessageReceived : function(  ) {},

/**Called when an {@link android.telephony.ims.ImsCallSession} may handover from one radio technology to another.
 For example, the session may handover from WIFI to LTE if conditions are right.
 <p>
 If handover is attempted,
 {@link #callSessionHandover(ImsCallSession, int, int, android.telephony.ims.ImsReasonInfo)} or
 {@link #callSessionHandoverFailed(ImsCallSession, int, int, android.telephony.ims.ImsReasonInfo)} will be
 called to indicate the success or failure of the handover.
@param {Object {ImsCallSession}} session IMS session object
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
*/
callSessionMayHandover : function(  ) {},

/**Called when session access technology changes
@param {Object {ImsCallSession}} session IMS session object
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
@param {Object {ImsReasonInfo}} reasonInfo
*/
callSessionHandover : function(  ) {},

/**Called when session access technology change fails
@param {Object {ImsCallSession}} session IMS session object
@param {Number} srcAccessTech original access technology
@param {Number} targetAccessTech new access technology
@param {Object {ImsReasonInfo}} reasonInfo handover failure reason
*/
callSessionHandoverFailed : function(  ) {},

/**Called when TTY mode of remote party changed
@param {Object {ImsCallSession}} session IMS session object
@param {Number} mode TTY mode of remote party
*/
callSessionTtyModeReceived : function(  ) {},

/**Notifies of a change to the multiparty state for this {@code ImsCallSession}.
@param {Object {ImsCallSession}} session The call session.
@param {Boolean} isMultiParty {@code true} if the session became multiparty, {@code false}
      otherwise.
*/
callSessionMultipartyStateChanged : function(  ) {},

/**Called when the session supplementary service is received
@param {Object {ImsCallSession}} session the session object that carries out the IMS session
*/
callSessionSuppServiceReceived : function(  ) {},

/**Received RTT modify request from Remote Party
*/
callSessionRttModifyRequestReceived : function(  ) {},

/**Received response for RTT modify request
*/
callSessionRttModifyResponseReceived : function(  ) {},

/**Device received RTT message from Remote UE
*/
callSessionRttMessageReceived : function(  ) {},

/**While in call, there has been a change in RTT audio indicator.
*/
callSessionRttAudioIndicatorChanged : function(  ) {},

/**Called when the IMS service reports a change to the call quality.
*/
callQualityChanged : function(  ) {},


};
