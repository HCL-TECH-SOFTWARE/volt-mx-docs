 Typeface.Builder buidler = new Typeface.Builder(getAssets(), "your_font_file.ttc");

 Typeface typeface = builder.build();
 </code>
 </pre>

 Typeface.Builder buidler = new Typeface.Builder("your_font_file.ttf");
