/**@class android.media.MediaPlayer2.EventCallback
@extends java.lang.Object

 Interface definition for callbacks to be invoked when the player has the corresponding
 events.
*/
var EventCallback = {

/**Called to indicate the video size

 The video size (width and height) could be 0 if there was no video,
 or the value was not determined yet.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 associated with this callback
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Object {Size}} size the size of the video
*/
onVideoSizeChanged : function(  ) {},

/**Called to indicate an avaliable timed text
@param {Object {MediaPlayer2}} mp the MediaPlayer2 associated with this callback
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Object {TimedText}} text the timed text sample which contains the text
             needed to be displayed and the display format.
@hide 
*/
onTimedText : function(  ) {},

/**Called to indicate avaliable timed metadata
 <p>
 This method will be called as timed metadata is extracted from the media,
 in the same order as it occurs in the media. The timing of this event is
 not controlled by the associated timestamp.
 <p>
 Currently only HTTP live streaming data URI's embedded with timed ID3 tags generates
 {@link android.media.TimedMetaData}.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 associated with this callback
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Object {TimedMetaData}} data the timed metadata sample associated with this event
@param mp the MediaPlayer2 associated with this callback
@param dsd the DataSourceDesc of this data source
@param data the timed metadata sample associated with this event
*/
onTimedMetaDataAvailable : function(  ) {},

/**Called to indicate an error.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 the error pertains to
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Number} what the type of error that has occurred.
@param {Number} extra an extra code, specific to the error. Typically
 implementation dependent.
*/
onError : function(  ) {},

/**Called to indicate an info or a warning.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 the info pertains to.
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Number} what the type of info or warning.
@param {Number} extra an extra code, specific to the info. Typically
 implementation dependent.
*/
onInfo : function(  ) {},

/**Called to acknowledge an API call.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 the call was made on.
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Number} what the enum for the API call.
@param {Number} status the returned status code for the call.
*/
onCallCompleted : function(  ) {},

/**Called to indicate media clock has changed.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 the media time pertains to.
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Object {MediaTimestamp}} timestamp the new media clock.
*/
onMediaTimeDiscontinuity : function(  ) {},

/**Called to indicate {@link #notifyWhenCommandLabelReached}(Object) has been processed.
@param {Object {MediaPlayer2}} mp the MediaPlayer2 {@link #notifyWhenCommandLabelReached(Object)} was called on.
@param {Object {Object}} label the application specific Object given by
        {@link #notifyWhenCommandLabelReached(Object)}.
*/
onCommandLabelReached : function(  ) {},

/**Called when when a player subtitle track has new subtitle data available.
@param {Object {MediaPlayer2}} mp the player that reports the new subtitle data
@param {Object {DataSourceDesc}} dsd the DataSourceDesc of this data source
@param {Object {MediaPlayer2.SubtitleData}} data the subtitle data
*/
onSubtitleData : function(  ) {},


};
