/**@class android.bluetooth.BluetoothHidDevice.Callback
@extends java.lang.Object

 The template class that applications use to call callback functions on events from the HID
 host. Callback functions are wrapped in this class and registered to the Android system
 during app registration.
*/
var Callback = {

/**Callback called when application registration state changes. Usually it's called due to
 either {@link android.bluetooth.BluetoothHidDevice#registerApp (String, String, String, byte, byte[],
 Executor, Callback)} or {@link android.bluetooth.BluetoothHidDevice#unregisterApp()} , but can be also
 unsolicited in case e.g. Bluetooth was turned off in which case application is
 unregistered automatically.
@param {Object {BluetoothDevice}} pluggedDevice {@link BluetoothDevice} object which represents host that currently
     has Virtual Cable established with device. Only valid when application is registered,
     can be <code>null</code>.
@param {Boolean} registered <code>true</code> if application is registered, <code>false</code>
     otherwise.
*/
onAppStatusChanged : function(  ) {},

/**Callback called when connection state with remote host was changed. Application can
 assume than Virtual Cable is established when called with {@link android.bluetooth.BluetoothProfile#STATE_CONNECTED} <code>state</code>.
@param {Object {BluetoothDevice}} device {@link BluetoothDevice} object representing host device which connection
     state was changed.
@param {Number} state Connection state as defined in {@link BluetoothProfile}.
*/
onConnectionStateChanged : function(  ) {},

/**Callback called when GET_REPORT is received from remote host. Should be replied by
 application using {@link android.bluetooth.BluetoothHidDevice#replyReport(BluetoothDevice, byte, byte,
 byte[])}.
@param {Object {BluetoothDevice}} type Requested Report Type.
@param {Number} id Requested Report Id, can be 0 if no Report Id are defined in descriptor.
@param {Number} bufferSize Requested buffer size, application shall respond with at least given
     number of bytes.
*/
onGetReport : function(  ) {},

/**Callback called when SET_REPORT is received from remote host. In case received data are
 invalid, application shall respond with {@link android.bluetooth.BluetoothHidDevice#reportError(BluetoothDevice, byte)}.
@param {Object {BluetoothDevice}} type Report Type.
@param {Number} id Report Id.
@param {Number} data Report data.
*/
onSetReport : function(  ) {},

/**Callback called when SET_PROTOCOL is received from remote host. Application shall use
 this information to send only reports valid for given protocol mode. By default, {@link android.bluetooth.BluetoothHidDevice#PROTOCOL_REPORT_MODE} shall be assumed.
@param {Object {BluetoothDevice}} protocol Protocol Mode.
*/
onSetProtocol : function(  ) {},

/**Callback called when report data is received over interrupt channel. Report Type is
 assumed to be {@link android.bluetooth.BluetoothHidDevice#REPORT_TYPE_OUTPUT}.
@param {Object {BluetoothDevice}} reportId Report Id.
@param {Number} data Report data.
*/
onInterruptData : function(  ) {},

/**Callback called when Virtual Cable is removed. After this callback is received connection
 will be disconnected automatically.
*/
onVirtualCableUnplug : function(  ) {},


};
